#+BEGIN_SRC haskell
import Data.List

notThe :: String -> Maybe String
notThe str = if str /= "the"
  then Just str
  else Nothing

replaceThe :: String -> String
replaceThe = unwords . fmap f . words
  where f xs = case notThe xs of
          (Just str) -> str
          Nothing    -> "a"
